C风格的函数指针
``` cpp
#include <iostream>

void printff(int input)
{
    std::cout << "printff " << input << '\n';
    return;
}

void function_tmp(void ptr(int))
{
    ptr(1);
}

int main()
{
    function_tmp(printff);
    return 0;
}
```
运行结果
``` cpp
$ ./a.out 
printff 1
```

lambda函数,这快具体看lambda部分的知识
``` cpp
#include <iostream>
#include <functional>

void printff(int input)
{
    std::cout << "printff " << input << '\n';
    return;
}

void function_tmp(void ptr(int))
{
    ptr(1);
}
void function_lambda(std::function<void(int)> ptr)
{
    ptr(2);
}

int main()
{
    int data = 10;
    function_tmp(printff);

    function_tmp([](int puter)
                 { std::cout << "puter " << puter << '\n'; });

    /* 错误的，当lambda函数有捕获时，必须使用function_lambda */
    // function_tmp([=](int puter)
    //              { std::cout << "data " << data << "  puter " << puter << '\n'; });

    function_lambda([=](int puter)
                    { std::cout << "data " << data << "  puter " << puter << '\n'; });
    return 0;
}
```
运行结果
``` cpp
$ ./a.out 
printff 1
puter 1
data 10  puter 2
```


类中函数指针的使用示例

实现了外部函数调用类的成员函数，成员函数内部自动切换不同的成员函数

``` cpp
#include <iostream>
#include <unistd.h>//linux下用来调用sleep函数延时的

class Box
{
public:
    int count;

    typedef void (Box::*BoxFuncPtr)();
    // using BoxFuncPtr = void (Box::*)();
    BoxFuncPtr func_ptr = nullptr;
    void BoxFun1();
    void BoxFun2();

    Box()
    {
        this->func_ptr = &Box::BoxFun1;
        this->count = 0;
    }

    void box_main_fun()
    {
        if (func_ptr)
            (this->*func_ptr)();
    }
};

void Box::BoxFun1()
{
    std::cout << "this is BoxFun1" << std::endl;
    this->func_ptr = &Box::BoxFun2;
    this->count++;
}

void Box::BoxFun2()
{
    std::cout << "this is BoxFun2" << std::endl;
    this->func_ptr = &Box::BoxFun1;
}

int main()
{
    Box *Box1;
    Box1 = new Box();
    std::cout << "main start" << std::endl;
    while (1)
    {
        Box1->box_main_fun();
        sleep(1);
        if (Box1->count > 5)
            break;
    }
    std::cout << "main end" << std::endl;
    delete Box1;
    Box1 = nullptr;

    return 0;
}
```

运行结果

this is BoxFun2
this is BoxFun1
this is BoxFun2

this is BoxFun2

this is BoxFun2
this is BoxFun1
main end
