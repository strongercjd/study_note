类中函数指针的使用示例

```
#include <iostream>
#include <unistd.h>

class Box
{
public:
    int count;

    typedef void (Box::*BoxFuncPtr)();
    // using BoxFuncPtr = void (Box::*)();
    BoxFuncPtr func_ptr = nullptr;
    void BoxFun1();
    void BoxFun2();

    Box()
    {
        this->func_ptr = &Box::BoxFun1;
        this->count = 0;
    }

    void box_main_fun()
    {
        if (func_ptr)
            (this->*func_ptr)();
    }
};

void Box::BoxFun1()
{
    std::cout << "this is BoxFun1" << std::endl;
    this->func_ptr = &Box::BoxFun2;
    this->count++;
}

void Box::BoxFun2()
{
    std::cout << "this is BoxFun2" << std::endl;
    this->func_ptr = &Box::BoxFun1;
}

int main()
{
    Box *Box1;
    Box1 = new Box();
    std::cout << "main start" << std::endl;
    while (1)
    {
        Box1->box_main_fun();
        sleep(1);
        if (Box1->count > 5)
            break;
    }
    std::cout << "main end" << std::endl;
    delete Box1;
    Box1 = nullptr;

    return 0;
}
```
